<script>
// 连接 Sepolia 测试网 RPC
const web3 = new Web3("https://rpc.sepolia.org");
let myWallet = null;

// 创建钱包（生成真实私钥）
async function createWallet(){
    myWallet = web3.eth.accounts.create();
    alert("钱包创建成功！\n地址："+myWallet.address+"\n私钥："+myWallet.privateKey+"\n请务必保存！");
    updateUI();
}

// 导入钱包
function importWallet(){
    document.getElementById("modalImport").classList.remove("hidden");
}
function doImport(){
    const pk = document.getElementById("pkInput").value;
    myWallet = web3.eth.accounts.privateKeyToAccount(pk);
    closeAll();
    updateUI();
}

// 刷新余额
async function refreshBalance(){
    if(!myWallet) return alert("请先创建/导入钱包");
    const bal = await web3.eth.getBalance(myWallet.address);
    document.getElementById("balance").innerText = web3.utils.fromWei(bal, "ether");
}

// 更新界面
function updateUI(){
    document.getElementById("address").innerText = myWallet.address;
    refreshBalance();
}

// 转账
function openTransfer(){
    if(!myWallet) return alert("请先创建/导入钱包");
    document.getElementById("modalTransfer").classList.remove("hidden");
}
async function sendTx(){
    const to = document.getElementById("toAddr").value;
    const amt = web3.utils.toWei(document.getElementById("amount").value, "ether");
    const tx = await web3.eth.sendTransaction({
        from: myWallet.address,
        to: to,
        value: amt,
        chainId: 11155111
    }).sign(myWallet.privateKey);
    alert("交易已发送，哈希："+tx.transactionHash);
    closeAll();
}

function closeAll(){
    document.querySelectorAll('[id^="modal"]').forEach(e=>e.classList.add("hidden"));
}
</script>
